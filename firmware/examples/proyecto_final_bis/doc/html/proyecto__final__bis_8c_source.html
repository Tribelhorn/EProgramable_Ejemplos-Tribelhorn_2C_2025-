172void leer_info(uint8_t * data, uint8_t length){
173 //Lee el horario de toma en formato (P-casilla-horario militar) = (P-CC-HHMM)/
174 //O setea la hora actual como (A-horario militar) = (A-HHMM) //
175 if (data [0] == LETRA_A && length >= 6) {
176 uint16_t hora_actual = (data[2]-NUMERO_0)*10 + (data[3]-NUMERO_0); //Se le resta el 0 en ASCII
177 uint16_t min_actual = (data[4]-NUMERO_0)*10 + (data[5]-NUMERO_0);
178 fecha_hora.hour = hora_actual;
179 fecha_hora.min = min_actual;
180 RtcConfig(&fecha_hora);
181 BleSendString("Hora configurada \n");
182 }
183
184 else if (data[0] == LETRA_P && length >= 9) {
185 uint16_t casilla = (((data[2]-NUMERO_0)*10)+(data[3]-NUMERO_0)-1);
186 uint16_t hora_pastilla = ( ((data[5]-NUMERO_0)*1000) + ((data[6]-NUMERO_0)*100) + ((data[7]-NUMERO_0)*10) + ((data[8]-NUMERO_0)) );
187 casillas[casilla] = hora_pastilla;
188 BleSendString("Horario recibido \n");
189 }
190}
